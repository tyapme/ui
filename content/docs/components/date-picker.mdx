```tsx
  DatePicker,
  DateRangePicker,
  DateFieldInput,
  TimeFieldInput,
} from "@/components/ui/date-picker"
```

```tsx
<DatePicker value={date} onValueChange={setDate} />
```

## 構成

```text
DatePicker
├── DateFieldInput (セグメント型入力)
│   └── DateSegment (年/月/日/時/分/秒)
└── Calendar Popover (カレンダー)
```

## API

### DatePicker

| Prop            | Type                                      | Default | Description                                           |
| --------------- | ----------------------------------------- | ------- | ----------------------------------------------------- |
| `value`         | `Date \| undefined`                       | -       | 選択された日付                                        |
| `onValueChange` | `(date: Date \| undefined) => void`       | -       | 日付変更時のコールバック                              |
| `showCalendar`  | `boolean`                                 | `true`  | カレンダーポップオーバーの表示/非表示                 |
| `granularity`   | `"day" \| "hour" \| "minute" \| "second"` | `"day"` | 入力の粒度。`"hour"` 以上で時刻セグメントが追加される |
| `disabled`      | `boolean`                                 | `false` | 無効状態                                              |
| `calendarProps` | `CalendarProps`                           | -       | Calendar に渡す追加 props                             |
| `className`     | `string`                                  | -       | className                                             |
| `id`            | `string`                                  | -       | ID                                                    |

### DateRangePicker

| Prop             | Type                                      | Default | Description                           |
| ---------------- | ----------------------------------------- | ------- | ------------------------------------- |
| `value`          | `DateRange \| undefined`                  | -       | 選択された日付範囲                    |
| `onValueChange`  | `(range: DateRange \| undefined) => void` | -       | 範囲変更時のコールバック              |
| `showCalendar`   | `boolean`                                 | `true`  | カレンダーポップオーバーの表示/非表示 |
| `disabled`       | `boolean`                                 | `false` | 無効状態                              |
| `numberOfMonths` | `number`                                  | `2`     | 表示する月数                          |
| `calendarProps`  | `CalendarProps`                           | -       | Calendar に渡す追加 props             |
| `className`      | `string`                                  | -       | className                             |
| `id`             | `string`                                  | -       | ID                                    |

### DateFieldInput

| Prop            | Type                                      | Default | Description              |
| --------------- | ----------------------------------------- | ------- | ------------------------ |
| `value`         | `Date \| undefined`                       | -       | 選択された日付           |
| `onValueChange` | `(date: Date \| undefined) => void`       | -       | 日付変更時のコールバック |
| `granularity`   | `"day" \| "hour" \| "minute" \| "second"` | `"day"` | 入力の粒度               |
| `disabled`      | `boolean`                                 | `false` | 無効状態                 |
| `className`     | `string`                                  | -       | className                |
| `id`            | `string`                                  | -       | ID                       |

### TimeFieldInput

| Prop            | Type                                | Default    | Description              |
| --------------- | ----------------------------------- | ---------- | ------------------------ |
| `value`         | `Date \| undefined`                 | -          | 時刻を含む Date          |
| `onValueChange` | `(date: Date \| undefined) => void` | -          | 時刻変更時のコールバック |
| `granularity`   | `"hour" \| "minute" \| "second"`    | `"minute"` | 入力の粒度               |
| `disabled`      | `boolean`                           | `false`    | 無効状態                 |
| `className`     | `string`                            | -          | className                |
| `id`            | `string`                            | -          | ID                       |

## キーボード操作

| キー        | 動作                       |
| ----------- | -------------------------- |
| `Tab`       | 次のセグメントに移動       |
| `Shift+Tab` | 前のセグメントに移動       |
| `↑` / `↓`   | セグメントの値を増減       |
| `0-9`       | セグメントに数値を直接入力 |
| `Backspace` | セグメントの値をクリア     |
